\chapter*{List of Acronyms}
\addcontentsline{toc}{chapter}{List of Acronyms}


% below is the list of acronym definitions, place them in alphabetical order
% since they will not be sorted again. 
\begin{acronym}

\acro{AC}{Aerogel Cherenkov}

\acro{ACC}{Air-Cooled Chiller}.

\acro{APRA}{Astrophysics Research and Analysis}. 

\acro{ASIC}{application specific integrated circuit}.

\acro{BOM}{Bill of Materials}.

\acro{CSBF}{Columbia Scientific Balloon Facility}. 

\acro{CRs}{Cosmic-Rays}.

\acro{CSP}{Charge Sensitive Pre-amp}.
    An electronics device that collect charge signal and amplify it into voltage signal.

\acro{COM}{Center of Mass}

\acro{CZT}{Cadmium Zinc Telluride}

\acro{ad}[$\mathbf{\overline{d}}$]{Antideuteron}.

\acro{DM}{Dark Matter}.

\acro{GAPS}{General Antiparticle Spectrometer}.
	The General Antiparticle Spectrometer (GAPS) is a current-generation mission to measure low-energy cosmic-ray antinuclei during at least three ~35-day Antarctic flights.

\acro{DAQ}{Data Acquisition}.

\acro{EBay}{Electronics Bay}.

\acro{ESS}{Electrostatic Separator}

\acro{FWHM}{Full Width at Half Maximum}.

\acro{GCS}{Ground Cooling System}.
    GAPS mission relies on pre-testing of all detector modules on the ground while under similar flight condition. To achieve strasomphere temperature conditions, GCS is built to cool down the detectors and dissipate heat generated on the ground.
    
\acro{GFP}{GAPS Functional Prototype}.
    A prototype of GAPS that is assembled and tested at the Bates Research and Engineering Center, verified concept of GAPS experiment.

\acro{GRAMS}{Gamma-Ray and Antimatter Survey}. 
    GRAMS is a novel project that can simultaneously target both astrophysical observations with MeV gamma rays and an indirect dark matter search with antimatter detection.

\acro{GSFC}{Goddard Space Flight Center}.

\acro{aHe3}[$\mathbf{\overline{He}_3}$]{Antihelium-3}

\acro{aHe4}[$\mathbf{\overline{He}_4}$]{Antihelium-4}

\acro{HK}{Housekeeping}.

\acro{ISM}{Interstella Medium}.

\acro{JFET}{The junction field-effect transistor}.

\acro{LAr}{Liquid Argon}.

\acro{LArTPC}{Liquid Argon Time Projection Chamber}.

\acro{LDB}{Long Duration Balloon}.

\acro{LLD}{Liquid Level Detector}.

\acro{MCHP}{Multi-Loop Capillary Heat Pipes}. Equivalent to OHP.

\acro{MIP}{Minimum Ionization Particle}

\acro{MPD}{Multiplicity Detector}

\acro{NTS}{NTS Technical Systems}. Institute that provide thermal vacuum test chamber.

\acro{OHP}{Oscillating Heat Pipe}. Equivalent to MCHP Unique noble thermal system that is designed for GAPS balloon flight that provide cooling power without an active pump.

\acro{ap}[$\mathbf{\overline{p}}$]{Antiproton}.

\acro{PCB}{Printed Circuit Board}.

\acro{PCIe}{Peripheral Component Interconnect Express}

\acro{PDE}{photon detection efficiency}

\acro{PDU}{Power Distribution Unit}. Usually balloon experiment has a power distribution design to control and power each device on payload during flight.

\acro{PMT}{Photomultiplier tube}.

\acro{pps}{pulse per second}

\acro{R23}{Trifluoromethane}. Refrigerant that is tested and optimized for GAPS flight thermal system.

\acro{ROI}{Region of Interest}.

\acro{RTD}{Resistance Temperature Detector}.

\acro{Si(Li)}{Lithium-Drifted Silicon detector}.

\acro{SiPM}{Silicon photomultiplier}.

\acro{SN}{Signal to Noise}.

\acro{SSL}{The Space Sciences Laboratory}. Lab at University of California, Berkeley

\acro{TOA}{Top of atmosphere}.

\acro{TOF}{Tim of Flight}. For a typical scitilation crystal system with fixed position, a recorded timing would behave as a time of flight detector system. This technique is used in both GAPS and GRAMS experiments.

\acro{TOI}{Top of instrument}.

\acro{TPB}{tetraphenyl-butadiene}

\acro{TVAC}{Thermal Vacuum}. Devices that exposed during balloon flight need to be tested under thermal vacuum condition.

\acro{WIMP}{Weakly interacting massive particle}

\acro{XMIT}{Transmitter board}

\end{acronym}
